#include <bits/stdc++.h>

using namespace std;

//Frog 2

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
   int n, k; 
   cin>>n>>k;
   int dp[n+1], heights[n+1]; 

   for(int i=1;i<=n;++i){
       cin>>heights[i];
       dp[i]=200000000;
   }
    dp[1]=0;

   
   for(int i=2;i<=n;++i){
       for(int j=1;j<=k;++j){
           if(i-j>0){
               dp[i]=min(dp[i], dp[i-j]+abs(heights[i]-heights[i-j]));
           }
       }
   }

   cout<<dp[n];
}
