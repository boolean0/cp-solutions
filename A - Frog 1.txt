#include <bits/stdc++.h>

using namespace std;


//Frog 1

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
   int n; 
   cin>>n;
   int dp[n+1], heights[n+1]; 

   for(int i=1;i<=n;++i){
       cin>>heights[i];
   }
    dp[1]=0;
    dp[2]=abs(heights[2]-heights[1]);
   for(int i=3;i<=n;++i){
       dp[i]=min(dp[i-1]+abs(heights[i]-heights[i-1]), dp[i-2]+abs(heights[i]-heights[i-2]));
   }

   cout<<dp[n];
}
